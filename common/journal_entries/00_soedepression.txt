je_soedepressionger = {

group = je_group_historical_content
    icon = "gfx/interface/icons/event_icons/depression.dds"
    is_shown_when_inactive = {
        c:GER = ROOT
    }
    possible = {
		has_variable = soe_depression_1_end_flag
    }
    complete = {
       any_scope_pop = {
				standard_of_living > 55
			}
    }
    on_complete = {
        add_modifier = { name = modifier_francogermandepressionfix
	 years = 5 }
    }
	
	should_be_pinned_by_default = yes
}
