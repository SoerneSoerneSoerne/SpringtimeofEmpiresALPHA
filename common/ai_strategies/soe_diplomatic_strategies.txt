ai_strategy_unify_china = {
	icon = "gfx/interface/icons/ai_strategy_icons/chinaunification.dds"
	
	type = diplomatic
		
	undesirable_infamy_level = {
		value = 100
	}
	
	unacceptable_infamy_level = {
		value = 200
	}
	
	ideological_opinion_effect_mult = {
		value = 0.5
	}
	
	wargoal_maneuvers_fraction = {
		value = 0.1
	}		
	
	diplomatic_play_neutrality = {
		value = 0
	}
	
	diplomatic_play_boldness = {
		value = 100
	}
	
	obligation_value = {
		value = 0
	
		if = {
			limit = { 
				scope:target_country = {
					OR = {
						country_has_primary_culture = cu:han
					}				
				}
			}
			add = 25
		}
	}		
	
	recklessness = {
		value = 2
	}	
	
	aggression = {
		value = 0.25
		
		if = {
			limit = { 
				scope:target_country = {
					OR = {
						country_has_primary_culture = cu:han
					}	
				}
			}
			add = 50
		}

		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}					
			}
			multiply = 1.5
		}

		if = {
			limit = {
				NOT = {
					is_adjacent = scope:target_country
				}
			}
			multiply = 0
		}			
	}

	building_group_weights = {
		bg_army = 2.5
	}
	
	wargoal_scores = {
		make_puppet = 0.0
		make_dominion = 0.0
		make_vassal = 0.0
		conquer_state = {
			if = {
				limit = {
					scope:target_state = {
						state_region = {
							OR = {
								is_homeland = cu:han
							}
						}
						owner = {
							NOR = { 
								is_supporting_unification_candidate = { who = root country_formation = CHI }
								is_subject_of = root
								has_diplomatic_pact = {
									who = scope:target_country
									type = customs_union
								}							
							}						
						}						
					}	
					can_reach_target_state = yes
				}
				add = 50
			}
		}
		return_state = {
			if = {
				limit = {
					scope:target_state = {
						state_region = {
							OR = {
								is_homeland = cu:han
							}
						}					
					}
					can_reach_target_state = yes
				}
				add = 200
			}
		}
		unification_leadership = {
			value = 1000
		}			 
	}	
	
	secret_goal_scores = {
		protect = {		
			if = {
				limit = {
					scope:target_country = {
						OR = {
							country_has_primary_culture = cu:han
						}
						NOT = { is_unification_candidate = CHI }
						OR = { 
							is_supporting_unification_candidate = { who = root country_formation = CHI }
							is_subject_of = root
							has_diplomatic_pact = {
								who = scope:target_country
								type = customs_union
							}							
						}	
					}
				}
				add = 1000
			}		
		}		
		dominate = {		
			if = {
				limit = {
					scope:target_country = {
						OR = {
							country_has_primary_culture = cu:han
						}					
						NOR = { 
							is_unification_candidate = CHI
							is_supporting_unification_candidate = { who = root country_formation = CHI }
							is_subject_of = root
							has_diplomatic_pact = {
								who = scope:target_country
								type = customs_union
							}							
						}	
					}
				}
				add = 1000
			}		
		}		
		conquer = {				
			if = {
				limit = {
					scope:target_country = {				
						country_has_primary_culture = cu:han
					}
					can_reach_target_country = yes
				}
				add = 1000
			}			
		}		
	}
		
	possible = {
		NOT = { country_tier = empire }
		country_has_primary_culture = cu:han
	}

	weight = {
		value = 1000	
	}
}
