ai_strategy_unify_india = {
	icon = "gfx/interface/icons/ai_strategy_icons/indiaunification.dds"
	
	type = diplomatic
	
	undesirable_infamy_level = {
		value = 100
	}
	
	unacceptable_infamy_level = {
		value = 200
	}
	
	ideological_opinion_effect_mult = {
		value = 0.5
	}
	
	wargoal_maneuvers_fraction = {
		value = 0.2
	}		
	
	diplomatic_play_neutrality = {
		value = 0
	}
	
	diplomatic_play_boldness = {
		value = 150
	}

	obligation_value = {
		value = 0
	
		if = {
			limit = { 
				scope:target_country = {			
					country_has_primary_culture = cu:avadhi
				}
			}
			add = 25
		}
	}	

	recklessness = {
		value = 2.0
	}
	
	aggression = {
		value = 0.25
		
		if = {
			limit = { 
				scope:target_country = {			
					country_has_primary_culture = cu:avadhi
				}
			}
			add = 50
		}

		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}					
			}
			multiply = 1.5
		}	
		if = {
			limit = {
				NOT = {
					is_adjacent_to_country = scope:target_country
				}
			}
			multiply = 0
		}		
	}

	building_group_weights = {
		bg_army = 2.5
	}
	
	wargoal_scores = {
		conquer_state = {
			if = {
				limit = {
					scope:target_state = {
						state_region = {
							is_homeland = cu:avadhi
						}
						owner = {
							NOR = { 
								is_supporting_unification_candidate = { who = root country_formation = BHT }
								is_subject_of = root
								has_diplomatic_pact = {
									who = scope:target_country
									type = customs_union
								}							
							}						
						}
					}
					can_reach_target_state = yes					
				}
				add = 100
			}
		}
		return_state = {
			if = {
				limit = {
					scope:target_state = {
						state_region = {
							is_homeland = cu:avadhi
						}	
					}
					can_reach_target_state = yes
				}
				add = 500
			}
		}		
		unification_leadership = {
			value = 1000
		}	
	}
	
	secret_goal_scores = {
		protect = {		
			if = {
				limit = {
					scope:target_country = {
						country_has_primary_culture = cu:avadhi
						NOT = { is_unification_candidate = BHT }
						OR = { 
							is_supporting_unification_candidate = { who = root country_formation = BHT }
							is_subject_of = root
							has_diplomatic_pact = {
								who = scope:target_country
								type = customs_union
							}							
						}						
					}
				}
				add = 1000
			}		
		}		
		dominate = {		
			if = {
				limit = {
					scope:target_country = {
						country_has_primary_culture = cu:avadhi				
						NOT = { is_unification_candidate = BHT }
						NOR = { 
							is_supporting_unification_candidate = { who = root country_formation = BHT }
							is_subject_of = root
							has_diplomatic_pact = {
								who = scope:target_country
								type = customs_union
							}							
						}	
					}
				}
				add = 1000
			}		
		}
		conquer = {				
			if = {
				limit = {
					scope:target_country = {				
						country_has_primary_culture = cu:avadhi
					}
					can_reach_target_country = yes
				}
				add = 1000
			}			
		}
	}	

	possible = {
		country_has_primary_culture = cu:avadhi	
		NOT = { country_tier = empire }
	}

	weight = {
		value = 1000
	}
}
