invade_denmark = {
	is_shown = {
		exists = c:SWE
		c:SWE = ROOT
		is_subject = no
		exists = c:DEN
	}

	possible = {
		army_size >= 75
		has_technology_researched = pan-nationalism
	}

	when_taken = {
        create_diplomatic_play = {     
        target_country = c:DEN
            type = dp_revolution
            war = yes
			}
	}
}
