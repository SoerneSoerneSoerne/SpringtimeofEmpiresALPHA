invade_walesbco = {
	is_shown = {
		exists = c:BCO
		c:BCO = ROOT
		is_subject = no
		exists = c:WLS
		NOT = { has_variable = britwar1 }
	}

	possible = {
		army_size >= 40
	}

	when_taken = {
	set_variable = britwar1
        create_diplomatic_play = {     
        target_country = c:WLS

			}
	}
	ai_chance = {
		base = 100		
	}
}
